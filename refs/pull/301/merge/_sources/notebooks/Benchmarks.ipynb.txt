    "            keras.layers.Conv2D(32, kernel_size=(3, 3), activation=\"relu\"),\n",

    "            keras.layers.Dropout(0.5),\n",
    "            keras.layers.Dense(num_classes, activation=\"softmax\"),\n",

    "    )\n",
